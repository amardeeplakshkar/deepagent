## Usage

```ts
import { RocketReachClient } from '@deepagent/rocketreach'

const rocketreach = new RocketReachClient()
const res = await rocketreach.lookupPerson({
  linkedin_url: 'https://www.linkedin.com/in/fisch2'
})
```
