## Usage

```ts
import { DuckDuckGoClient } from '@deepagent/duck-duck-go'

// No API is required to use the DuckDuckGo API
const duckDuckGo = new DuckDuckGoClient()
const results = await duckDuckGo.search({
  query: 'latest news about AI'
})
```
